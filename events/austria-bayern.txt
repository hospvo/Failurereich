
add_namespace = sgf

country_event = {
	id = sgf.1
	title = "Referundum about Austria"
	desc = "Prussian parlement try to find if they can integrate Austria, this referundum will be made because Petrograd alliance collapsed."
	picture = GFX_report_event_generic_read_write
	
	fire_only_once = yes

	trigger= {
		tag = BAY
		country_exists = AUS
		country_exists = GER
		NOT = { country_exists = WUR }
		BAY = { has_government = fascism }
        }
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { 
		name = "Propose referundum"
		AUS = {
			country_event = { days = 1 id = sgf.2 }
		}
	}
	option = { 
		name = "Invade Austria"
		
	        declare_war_on = {
		                target = AUS
		                type = annex_everything
	        }	
			AUS = {
				every_owned_state = {
				    add_claim_by = BAY 
				}
			}

	}
}
